class Solution {
public:
    int goodNodes(TreeNode* root) {
        return dfs(root, root->val);  // start DFS with initial max = root->val
    }

private:
    int dfs(TreeNode* node, int maxSoFar) {
        if (!node) return 0; // base case: no node means no good nodes here

        // check if this node is good
        int count = (node->val >= maxSoFar) ? 1 : 0;

        // update the maximum value seen so far
        int newMax = max(maxSoFar, node->val);

        // continue DFS to left and right
        count += dfs(node->left, newMax);
        count += dfs(node->right, newMax);

        return count;
    }
};
